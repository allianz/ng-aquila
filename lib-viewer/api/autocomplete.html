import { NxAutocompleteModule } from '@aposin/ng-aquila/autocomplete';

Directives

NxAutocompleteOptionComponent

Selector:

nx-autocomplete-option

Properties
Name Description
@Input()

disabled: boolean

Whether the option is disabled.
@Input()

value: any

The value of the autocomplete option.
@Output()

onSelectionChange: EventEmitter

Event emitted when the option is selected or deselected.

active: boolean

Whether or not the option is currently active and ready to be selected. An active option displays styles as if it is focused, but the focus is actually retained somewhere else. This comes in handy for components like autocomplete where focus must remain on the input.

id: string

Id of the autocomplete option. By default it is set to an incremented value.

selected: boolean

Whether or not the option is currently selected.
Methods
deselect()

Deselects the option.

select()

Selects the option.

NxAutocompleteTriggerDirective

Selector:

input[nxAutocomplete] textarea[nxAutocomplete]

Exported as: nxAutocompleteTrigger
Properties
Name Description
@Input('nxAutocomplete')

autocomplete: NxAutocompleteComponent

The autocomplete panel to be attached to this trigger.
@Input('nxAutocompleteDisabled')

autocompleteDisabled: boolean

Whether autocomplete functionality is disabled.
@Input('nxAutocompleteDebounce')

debounce: number

Debounce in ms before items callback is triggered. Defaults to 400
@Input('nxAutocompleteItems')

itemsCb: (val: string) => Observable

The items callback. Called with input value, must return Observable of Array of strings

activeOption: NxAutocompleteOptionComponent | null

The currently active option, coerced to NxAutocompleteOptionComponent type.

optionSelections: Observable

Stream of autocomplete option selections.

panelClosingActions: Observable

A stream of actions that should close the autocomplete panel, including when an option is selected, on blur, and when TAB is pressed.

panelOpen: boolean

Whether or not the autocomplete panel is open.
Methods
closePanel()

Closes the autocomplete suggestion panel.

openPanel()

Opens the autocomplete suggestion panel.

NxAutocompleteComponent

Selector:

nx-autocomplete

Exported as: nxAutocomplete
Properties
Name Description
@Input('class')

classList: string

Takes classes set on the host nx-autocomplete element and applies them to the panel inside the overlay container to allow for easy styling.
@Input()

items: string[] | Observable

Items observable. If given, autocomplete takes care of the options and ng-content is discarderd
@Input('nxValueFormatter')

valueFormatter: (value: any) => string

Value to string converter function. As an autocomplete option can hold any value, a converter might be needed to get a string representation of the value, which can be stored in the input.

Default: value ? value.toString() : null.
@Output()

closed: EventEmitter

Event that is emitted when the autocomplete panel is closed.
@Output()

opened: EventEmitter

Event that is emitted when the autocomplete panel is opened.
@Output()

optionSelected: EventEmitter

Event that is emitted whenever an option from the list is selected.

id: string

Unique ID to be used by autocomplete trigger's "aria-owns" property.

isOpen: boolean

Whether the autocomplete panel is open.

options: QueryList

Autocomplete options

showPanel: boolean

Whether the autocomplete panel should be visible, depending on option length.
