import { NxTooltipModule } from '@allianz/ng-aquila/tooltip';

Directives

NxTooltipDirective

Directive that attaches a tooltip to the host element.

Selector: [nxTooltip]

Exported as: nxTooltip

Properties
Name Description
Input{ alias: nxTooltipDisabled } @Input ('nxTooltipDisabled')

disabled: boolean

Disables the display of the tooltip.
Input{ alias: nxTooltipHideDelay } @Input ('nxTooltipHideDelay')

hideDelay: number

The default delay in ms before hiding the tooltip after hide is called
Input

manualTrigger: InputSignal<boolean>

Whether the tooltip should only be shown or hidden manually. If true, tooltip will not respond to mouse or focus events.
Input{ alias: nxTooltip } @Input ('nxTooltip')

message: string

The message to be displayed in the tooltip
Input{ alias: nxTooltipPosition } @Input ('nxTooltipPosition')

position: TooltipPosition

Allows the user to define the position of the tooltip relative to the parent element
Input{ alias: nxTooltipSelectable } @Input ('nxTooltipSelectable')

selectable: boolean

Allow selection of text within tooltip trigger.

NOTE: inputs and textareas always remain selectable, ignoring this input.
Input{ alias: nxTooltipShowDelay } @Input ('nxTooltipShowDelay')

showDelay: number

The default delay in ms before showing the tooltip after show is called
Methods
hide( delay: number = this.hideDelay )

Hides the tooltip after the delay in ms, defaults to tooltip-delay-hide or 0ms if no input

Parameters:

delay: number = this.hideDelay
show( delay: number = this.showDelay )

Shows the tooltip after the delay in ms, defaults to tooltip-delay-show or 0ms if no input

Parameters:

delay: number = this.showDelay
toggle()

Shows/hides the tooltip

Additional interfaces

NxTooltipDefaultOptions

Default nxTooltip options that can be overridden.

Properties
Name Description

hideDelay: number

The default delay in ms before hiding the tooltip after hide is called

position?: TooltipPosition

The default position of the tooltip

showDelay: number

The default delay in ms before showing the tooltip after show is called

touchendHideDelay: number

The default delay in ms before hiding the tooltip on touch devices

Additional type aliases

TooltipPosition

Possible positions of the tooltip.

Type: TooltipVerticalPosition | TooltipHorizontalPosition

TooltipHorizontalPosition

Type: 'left' | 'right'

TooltipVerticalPosition

Type: 'top' | 'bottom'
